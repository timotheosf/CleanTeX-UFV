\newcommand{\PrintCoAdvisors}[1]{
    \def\contador{0}%
    \def\unicoitem{}%
    % Executa o laço contabilizando os elementos
    \foreach \valor [count=\i] in {#1} {%
        \xdef\contador{\i}% Guarda o número de elementos globalmente
        \xdef\unicoitem{\valor}% Guarda o valor caso haja apenas um
    }%
    \ifnum\contador=1\relax
        \textbf{Coorientador:} \unicoitem
    \else
        \textbf{Coorientadores:}
        \foreach \nome [count=\i] in {#1} {%
            \nome \ifnum\i<\contador \\ \fi
        }%
    \fi
}
