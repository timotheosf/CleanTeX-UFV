\newcommand{\BuildTemplateSetup}{%
    
    %> 1. Carrega os pacotes e ferramentas do usuário
    \input{_preamble/packages}
    \input{_preamble/standard_notation}

    %> 2. Carrega as opções de formatação: UFV / Publish
    \iftoggle{ufvnorms}{%
        \input{_preamble/format2ufv}     % Carrega o driver da UFV
    }{%
        \input{_preamble/format2publish} % Carrega o driver fancy
    }

    %> 3. Carrega o hyperref e bibtex
    \usepackage{hyperref}
    \iftoggle{ufvnorms}{%
        \hypersetup{
            colorlinks=true, 
            allcolors=black
        }
    }{%
        \hypersetup{
            colorlinks=true,
        	linkcolor=NavyBlue,
        	urlcolor=MediumBlue,
        	citecolor=ForestGreen,
        	pdfhighlight=/N
        }

    }
    \addbibresource{_bibliography/references.bib}

    %> 4. Metadados no PDF e no Título do LaTeX
    \hypersetup{
        pdftitle={\thesistitle},
        pdfauthor={\thesisauthor}
    }
    \title{\thesistitle}
    \author{\thesisauthor}

    %> 5. Carregar os macros pré-textuais
    \input{_title/generalmacros}
