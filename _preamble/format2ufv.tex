%> Formatação dos capítulos enumerados alinhada à esquerda
\titleformat{\chapter}[block]
    {\normalfont\bfseries\filright}
    {\thechapter} % Numeração do capítulo em formato N
    {0.3cm} % Espaço em branco entre número e número do capítulo
    {\MakeUppercase} % Aplica uppercase ao nome do capítulo
    [\vspace{0.3cm}\titlerule] % Faz uma linha (rule) logo após o título

%> Formatação dos capítulos não-enumerados alinhada ao centro
\titleformat{name=\chapter,numberless}[block]
    {\normalfont\bfseries\filcenter}{}{0pt}
    {\MakeUppercase} % Aplica uppercase ao nome do capítulo
    [\vspace{0.3cm}\titlerule]

%> Configuração das epígrafes

  \begin{flushright}
  \begin{minipage}{.3\textwidth}
    \hspace{0.3cm}{#1}\vspace{0.3cm}\\
    {\itshape\small#2}
  \end{minipage}\hspace*{2cm}
\end{flushright}
\vspace{1.5cm}
}
